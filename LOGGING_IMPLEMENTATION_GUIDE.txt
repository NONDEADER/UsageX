# IMPLEMENTATION GUIDE: LOGGING SYSTEM FOR USAGEX

## Quick Start: Copy This Pattern

### 1. Core RawLog Function (utils.js)
`javascript
async function RawLog(sender, ...args) {
  let level = "debug";

  // Parse optional log level from first argument
  if (typeof args[0] === 'string' && ["debug", "warn", "error"].includes(args[0])) {
    level = args.shift();
  }

  // Check if debug mode is enabled
  const debugUntil = await getStorageValue('debug_mode_until');
  const now = Date.now();

  if ((!debugUntil || debugUntil <= now) && !FORCE_DEBUG) {
    return;  // Silent return if debug disabled
  }

  // Console output for developer debugging
  if (level === "debug") {
    console.log("[UsageX]", ...args);
  } else if (level === "warn") {
    console.warn("[UsageX]", ...args);
  } else if (level === "error") {
    console.error("[UsageX]", ...args);
  }

  // Format timestamp with milliseconds
  const timestamp = new Date().toLocaleString('default', {
    hour: '2-digit',
    minute: '2-digit',
    second: '2-digit',
    hour12: false,
    fractionalSecondDigits: 3
  });

  // Create log entry object
  const logEntry = {
    timestamp: timestamp,
    sender: sender,
    level: level,
    message: args.map(arg => {
      if (arg instanceof Error) {
        return arg.stack || \\: \\;
      }
      if (typeof arg === 'object') {
        if (arg === null) return 'null';
        try {
          return JSON.stringify(arg, Object.getOwnPropertyNames(arg), 2);
        } catch (e) {
          return String(arg);
        }
      }
      return String(arg);
    }).join(' ')
  };

  // Get existing logs, add new one, enforce limit
  const logs = await getStorageValue('debug_logs', []);
  logs.push(logEntry);

  // Circular buffer: keep only last 1000 entries
  if (logs.length > 1000) {
    logs.shift();  // Remove oldest
  }

  // Save back to storage
  await setStorageValue('debug_logs', logs);
}
`

### 2. Wrapper Function in Each Script
`javascript
// In background.js
async function Log(...args) {
  await RawLog("background", ...args);
}

// In content script
async function Log(...args) {
  const pageTitle = document.title.substring(0, 20);
  const sender = \content:\\;
  await RawLog(sender, ...args);
}
`

`javascript

  await Log("debug", "Large object:", largeData);
}

### 2. Controller Script (debug.js)
`javascript
let autoRefreshInterval;
const PERMANENT_DEBUG = 8640000000000000;

// Event listeners
document.getElementById('refresh').addEventListener('click', showLogs);
document.getElementById('clear').addEventListener('click', clearLogs);
document.getElementById('enableDebug').addEventListener('click', toggleDebugMode);
document.getElementById('enablePermanent').addEventListener('click', enablePermanentDebug);

async function showLogs() {
  const result = await browser.storage.local.get('debug_logs');
  const logs = result.debug_logs || [];
  const container = document.getElementById('logs');
  
  container.innerHTML = '';

  logs.forEach(log => {
    const line = document.createElement('div');
    line.className = 'log-line';
    line.dataset.level = log.level || 'debug';

    const timestamp = document.createElement('span');
    timestamp.className = 'log-timestamp';
    timestamp.textContent = log.timestamp;

    const sender = document.createElement('span');
    sender.className = 'log-sender';
    sender.dataset.sender = log.sender;
    sender.textContent = log.sender;

    const message = document.createElement('span');
    message.className = 'log-message';
    message.textContent = log.message;

    line.appendChild(timestamp);
    line.appendChild(sender);
    line.appendChild(message);
    container.appendChild(line);
  });

  container.scrollTop = container.scrollHeight;
}

async function clearLogs() {
  await browser.storage.local.set({ debug_logs: [] });
  showLogs();
}

async function updateDebugStatus() {
  const result = await browser.storage.local.get('debug_mode_until');
  const debugUntil = result.debug_mode_until;
  const now = Date.now();
  const isEnabled = debugUntil && debugUntil > now;
  const isPermanent = debugUntil === PERMANENT_DEBUG;
  const timeLeft = isEnabled && !isPermanent ? Math.ceil((debugUntil - now) / 60000) : 0;

  const statusEl = document.getElementById('debugStatus');
  statusEl.textContent = isPermanent
    ? 'Debug: Permanent'
    : isEnabled
      ? \Debug: \m remaining\
      : 'Debug: Disabled';

  const debugBtn = document.getElementById('enableDebug');
  const permBtn = document.getElementById('enablePermanent');

  if (isEnabled) {
    debugBtn.textContent = 'Disable Debug';
    permBtn.style.display = 'none';
    if (!autoRefreshInterval) startAutoRefresh();
  } else {
    debugBtn.textContent = 'Enable Debug (1h)';
    permBtn.style.display = 'inline-block';
    if (autoRefreshInterval) stopAutoRefresh();
  }
}

async function toggleDebugMode() {
  const result = await browser.storage.local.get('debug_mode_until');
  const debugUntil = result.debug_mode_until;
  const now = Date.now();
  const isEnabled = debugUntil && debugUntil > now;

  if (isEnabled) {
    await browser.storage.local.set({ debug_mode_until: now });
  } else {
    const oneHour = Date.now() + 60 * 60 * 1000;
    await browser.storage.local.set({ debug_mode_until: oneHour });
  }

  updateDebugStatus();
}

async function enablePermanentDebug() {
  await browser.storage.local.set({ debug_mode_until: PERMANENT_DEBUG });
  updateDebugStatus();
}

function startAutoRefresh() {
  if (!autoRefreshInterval) {
    autoRefreshInterval = setInterval(() => {
      if (document.getElementById('autoUpdate').checked) {
        showLogs();
      }
      updateDebugStatus();
    }, 5000);
  }
}

function stopAutoRefresh() {
  if (autoRefreshInterval) {
    clearInterval(autoRefreshInterval);
    autoRefreshInterval = null;
  }
}

// Initialize
showLogs();
updateDebugStatus();
startAutoRefresh();

window.addEventListener('beforeunload', stopAutoRefresh);
`

## Storage Helper Functions Needed

`javascript
async function getStorageValue(key, defaultValue = null) {
  const result = await browser.storage.local.get(key);
  return result[key] ?? defaultValue;
}

async function setStorageValue(key, value) {
  await browser.storage.local.set({ [key]: value });
}

async function isDebugMode() {
  const debugUntil = await getStorageValue('debug_mode_until');
  return debugUntil && debugUntil > Date.now();
}

async function enableDebugForHour() {
  const oneHour = Date.now() + 60 * 60 * 1000;
  await setStorageValue('debug_mode_until', oneHour);
}
`

## Performance Considerations

### Optimization Tips

1. **Avoid Logging in Tight Loops**
   `javascript
   // BAD
   for (let i = 0; i < 10000; i++) {
     await Log("Item", i);  // 10000 storage writes!
   }

   // GOOD
   const items = [];
   for (let i = 0; i < 10000; i++) {
     items.push(i);
   }
   await Log("Items processed:", items.length);
   `

2. **Check Debug Before Expensive Operations**
   `javascript
   const isDebug = await isDebugMode();
   if (isDebug) {
     await Log("Expensive object:", expensiveObjectStringification());
   }
   `

3. **Batch Storage Operations**
   `javascript
   // Instead of multiple Log calls, batch them
   const logs = [];
   logs.push("event1");
   logs.push("event2");
   logs.push("event3");
   // Then log once if needed
   `

4. **Consider Debouncing Auto-Refresh**
   - Current 5-second interval is reasonable
   - Increase if performance issues occur
   - Use equestAnimationFrame for render optimization

## Advanced Features to Consider

### 1. Log Filtering
`javascript
function filterLogs(logs, { sender, level, text }) {
  return logs.filter(log => {
    if (sender && !log.sender.includes(sender)) return false;
    if (level && log.level !== level) return false;
    if (text && !log.message.includes(text)) return false;
    return true;
  });
}
`

### 2. Log Export
`javascript
async function exportLogs(format = 'json') {
  const result = await browser.storage.local.get('debug_logs');
  const logs = result.debug_logs || [];
  
  let content;
  if (format === 'json') {
    content = JSON.stringify(logs, null, 2);
  } else if (format === 'csv') {
    content = 'timestamp,sender,level,message\n' +
      logs.map(l => 
        \"\","\","\","\"\
      ).join('\n');
  }
  
  const blob = new Blob([content], { type: 'text/plain' });
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = \logs-\.txt\;
  a.click();
}
`

### 3. Remote Logging
`javascript
async function sendLogsToServer(serverUrl) {
  const result = await browser.storage.local.get('debug_logs');
  const logs = result.debug_logs || [];
  
  try {
    await fetch(serverUrl, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        extensionId: chrome.runtime.id,
        timestamp: new Date().toISOString(),
        logs: logs
      })
    });
  } catch (error) {
    await Log("error", "Failed to send logs:", error);
  }
}
`

### 4. IndexedDB for Better Querying
`javascript
// For larger log collections
async function initLogDatabase() {
  return new Promise((resolve, reject) => {
    const request = indexedDB.open('UsageXLogs', 1);
    
    request.onerror = () => reject(request.error);
    request.onsuccess = () => resolve(request.result);
    
    request.onupgradeneeded = (event) => {
      const db = event.target.result;
      const store = db.createObjectStore('logs', { keyPath: 'id', autoIncrement: true });
      store.createIndex('sender', 'sender');
      store.createIndex('level', 'level');
      store.createIndex('timestamp', 'timestamp');
    };
  });
}
`
