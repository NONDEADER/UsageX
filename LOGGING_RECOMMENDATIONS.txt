# COMPARATIVE ANALYSIS: CLAUDE TRACKER vs ENHANCED USAGEX LOGGING

## Current Implementation Strengths (Claude Usage Tracker)

### ✓ Storage & Persistence
- Centralized browser.storage.local (IndexedDB)
- Automatic persistence across sessions
- Simple array-based structure
- 1000-entry circular buffer prevents bloat
- No external dependencies

### ✓ Multi-Source Logging
- Background script logging
- Content script logging
- Global error handlers (window.error, unhandledrejection)
- Error stack traces captured
- Per-component identification

### ✓ Debug Mode Control
- Toggle on/off with time-based expiration
- 1-hour default duration
- Permanent mode option (sentinel value)
- Countdown timer display
- Conditional logging (FORCE_DEBUG flag)

### ✓ Real-time UI
- Auto-refresh with checkbox control
- 5-second polling interval
- Color-coded by source and level
- Auto-scroll to bottom
- Clean dark theme (VS Code inspired)
- Responsive button controls

### ✓ Timestamp Precision
- Millisecond granularity
- Human-readable format (HH:MM:SS.mmm)
- Consistent formatting across all logs
- 24-hour time format

### ✓ Error Handling
- Stack trace preservation
- Safe JSON serialization with fallbacks
- Error object detection
- Null value handling
- Circular reference protection

async function applyFilters() {
  const result = await browser.storage.local.get('debug_logs');
  let logs = result.debug_logs || [];
  
  // Text search
  const searchText = document.getElementById('searchBox').value;
  if (searchText) {
    const regex = new RegExp(searchText, 'i');
    logs = logs.filter(log => 
      regex.test(log.message) || regex.test(log.sender)
    );
  }
  
  // Level filter
  const levels = [];
  if (document.getElementById('showDebug').checked) levels.push('debug');
  if (document.getElementById('showWarn').checked) levels.push('warn');
  if (document.getElementById('showError').checked) levels.push('error');
  logs = logs.filter(log => levels.includes(log.level));
  
  // Sender filter
  const sender = document.getElementById('senderFilter').value;
  if (sender) {
    logs = logs.filter(log => log.sender.includes(sender));
  }
  
  // Display filtered logs
  displayLogs(logs);
}

// Attach listeners
['searchBox', 'showDebug', 'showWarn', 'showError', 'senderFilter'].forEach(id => {
  document.getElementById(id).addEventListener('change', applyFilters);
});

async function generateAnalytics() {
  const result = await browser.storage.local.get('debug_logs');
  const logs = result.debug_logs || [];
  
  const analytics = {
    total: logs.length,
    byLevel: {
      debug: logs.filter(l => l.level === 'debug').length,
      warn: logs.filter(l => l.level === 'warn').length,
      error: logs.filter(l => l.level === 'error').length
    },
    bySender: {},
    byHour: {}
  };
  
  logs.forEach(log => {
    // Count by sender
    analytics.bySender[log.sender] = (analytics.bySender[log.sender] || 0) + 1;
    
    // Extract hour from timestamp
    const hour = log.timestamp.split(':')[0];
    analytics.byHour[hour] = (analytics.byHour[hour] || 0) + 1;
  });
  
  return analytics;
}

// Could use Chart.js for visualization

### 4. Database Migration to IndexedDB

**Claude Tracker: Uses browser.storage.local (simpler)**

**Recommended for UsageX (if logs grow large):**

`
Enhancement Level: MEDIUM
Effort: 4 hours
Priority: LOW (only if >10k logs needed)

Reasons for migration:
- Larger storage quota (typically >50MB)
- Better indexing for queries
- Asynchronous operations
- Structured data storage
- Better performance for large datasets

Downsides:
- More complex implementation
- Harder to debug
- Overkill for current needs
`

**Simple IndexedDB Wrapper**:
`javascript
class LogDatabase {
  constructor() {
    this.db = null;
  }
  
  async init() {
    return new Promise((resolve, reject) => {
      const request = indexedDB.open('UsageXLogs', 1);
      
      request.onerror = () => reject(request.error);
      request.onsuccess = () => {
        this.db = request.result;
        resolve();
      };
      
      request.onupgradeneeded = (event) => {
        const db = event.target.result;
        if (!db.objectStoreNames.contains('logs')) {
          const store = db.createObjectStore('logs', 
            { keyPath: 'id', autoIncrement: true }
          );
          store.createIndex('sender', 'sender');
          store.createIndex('level', 'level');
          store.createIndex('timestamp', 'timestamp');
        }
      };
    });
  }
  
  async addLog(logEntry) {
    const tx = this.db.transaction('logs', 'readwrite');
    const store = tx.objectStore('logs');
    return new Promise((resolve, reject) => {
      const request = store.add(logEntry);
      request.onerror = () => reject(request.error);
      request.onsuccess = () => resolve(request.result);
    });
  }
  
  async query(filter) {
    const tx = this.db.transaction('logs', 'readonly');
    const store = tx.objectStore('logs');
    
    if (filter.level) {
      const index = store.index('level');
      return await this._rangeQuery(index.getAll(filter.level));
    }
    
    return await this._rangeQuery(store.getAll());
  }
  
  async _rangeQuery(request) {
    return new Promise((resolve, reject) => {
      request.onerror = () => reject(request.error);
      request.onsuccess = () => resolve(request.result);
    });
  }
}
`

`javascript
async function sendLogsToServer(endpoint, batchSize = 100) {
  const result = await browser.storage.local.get('debug_logs');
  const logs = result.debug_logs || [];
  
  // Send in batches to avoid overwhelming server
  for (let i = 0; i < logs.length; i += batchSize) {
    const batch = logs.slice(i, i + batchSize);
    
    try {
      const response = await fetch(endpoint, {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'X-Extension-Version': chrome.runtime.getManifest().version
        },
        body: JSON.stringify({
          extensionId: chrome.runtime.id,
          timestamp: new Date().toISOString(),
          logs: batch,
          userAgent: navigator.userAgent
        })
      });
      
      if (response.ok) {
        await Log("Successfully sent batch", i / batchSize + 1);
      }
    } catch (error) {
      await Log("error", "Failed to send logs:", error);
    }
  }
}
`

### 6. Persistent Log Archiving

**Claude Tracker: NOT IMPLEMENTED**

**Recommended for UsageX:**

`
Enhancement Level: MEDIUM
Effort: 2 hours
Priority: MEDIUM

Features:
- Archive logs by date
- Keep rolling 7-day history
- Compress archived logs
- Easy access to previous sessions
- Automatic cleanup of old archives

Implementation:
- Create log_archive_YYYY-MM-DD keys
- Archive when current reaches 1000 entries
- Timestamp-based retention policy
`

**Implementation**:
`javascript
async function archiveCurrentLogs() {
  const today = new Date().toISOString().split('T')[0];
  const archiveKey = \debug_logs_archive_\\;
  
  const result = await browser.storage.local.get('debug_logs');
  const currentLogs = result.debug_logs || [];
  
  if (currentLogs.length === 0) return;
  
  // Get existing archive for today
  const archiveResult = await browser.storage.local.get(archiveKey);
  let archivedLogs = archiveResult[archiveKey] || [];
  
  // Merge and save
  archivedLogs = [...archivedLogs, ...currentLogs];
  await browser.storage.local.set({
    [archiveKey]: archivedLogs,
    debug_logs: []  // Clear current
  });
  
  // Cleanup old archives (older than 30 days)
  const allKeys = await browser.storage.local.get(null);
  const thirtyDaysAgo = new Date();
  thirtyDaysAgo.setDate(thirtyDaysAgo.getDate() - 30);
  
  Object.keys(allKeys).forEach(key => {
    if (key.startsWith('debug_logs_archive_')) {
      const dateStr = key.replace('debug_logs_archive_', '');
      const archiveDate = new Date(dateStr);
      if (archiveDate < thirtyDaysAgo) {
        browser.storage.local.remove(key);
      }
    }
  });
}

// Call when logs reach 1000 or on timer
setInterval(archiveCurrentLogs, 24 * 60 * 60 * 1000); // Daily
`
